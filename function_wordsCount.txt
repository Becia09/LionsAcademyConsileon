public int wordsCount(String[] words, int len) {
  
    int howManyWords = 0;
    for (String word : words)
    {
        int howManyLetters = 0;
        for (char letter : word.toCharArray())
        {
            howManyLetters++;
            if (howManyLetters > len)
            {
                break;
            }
        }
        if (howManyLetters == len)
        {
            howManyWords++;
        }
    }
    
    return howManyWords;
}
